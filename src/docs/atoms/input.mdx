import { Playground, Props } from 'docz'
import { Input, DefaultInput, StandardInput, OutlinedInput, RoundedInput } from '@components/atoms'

# Input

<Props of={Input} />

## Basic usage

Input box with no animations.

<Playground>
    <Input type="text" placeholder="name" name="name" />
</Playground>

## DefaultInput

Input with border animation.

<Playground>
    <DefaultInput type="name" placeholder="name" name="name" />
</Playground>

## 

**error** prop is used to mark the input as error.

<Playground>
    <DefaultInput type="password" placeholder="password" name="password" error />
</Playground>

## StandardInput

`onChange` property is used to set the user input.

<Playground>
    <StandardInput type="text" placeholder="name" name="name" width="250px" onChange={() => alert("Change triggered")}/>
</Playground>
